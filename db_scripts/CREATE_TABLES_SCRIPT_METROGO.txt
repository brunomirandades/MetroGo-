CREATE TABLE USERS(ID_USER NUMERIC(5) NOT NULL,
LOGIN VARCHAR(15) NOT NULL,
PASSWORD VARCHAR(15) NOT NULL,
PRIMARY KEY(ID_USER));

CREATE TABLE ADMINISTRADORAS(ID_ADM NUMERIC(4) NOT NULL,
NOME_ADM VARCHAR(50) NOT NULL,
PRIMARY KEY(ID_ADM));

CREATE TABLE LINHAS(ID_LINHA NUMERIC(4) NOT NULL,
NOME_LINHA VARCHAR(50) NOT NULL,
ID_ADM NUMERIC(4) NOT NULL,
PRIMARY KEY(ID_LINHA),
FOREIGN KEY(ID_ADM) REFERENCES ADMINISTRADORAS(ID_ADM));

CREATE TABLE ESTACOES(ID_ESTACAO NUMERIC(9) NOT NULL,
NOME_ESTACAO VARCHAR(50) NOT NULL,
TRANSF DOUBLE NOT NULL,
PRIMARY KEY(ID_ESTACAO));

CREATE TABLE LINHAS_ESTACOES(POSICAO NUMERIC(11) NOT NULL,
NOME_CONEXAO VARCHAR(50) NOT NULL,
ID_LINHA NUMERIC(4) NOT NULL,
ID_ESTACAO NUMERIC(9) NOT NULL,
PRIMARY KEY(ID_LINHA, ID_ESTACAO),
FOREIGN KEY(ID_LINHA) REFERENCES LINHAS(ID_LINHA),
FOREIGN KEY(ID_ESTACAO) REFERENCES ESTACOES(ID_ESTACAO));
